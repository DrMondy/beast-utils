# isObjectValEqual
对比两个对象的值是否相等（只支持基础数据类型的比较）

### 参数

```typescript
/**
 * 对比两个对象的值是否相等（只支持基础数据类型的比较）needCheckAttrs (需要比较的属性 key),默认比较全部
 *@param srcObj<Object>
 *@param othObj<Object>
 *@param needCheckAttrs<Array>
 */
const isObjectValEqual = (srcObj,othObj,needCheckAttrs=null) => boolean;
```

### 实例

```javascript
let srcObj = {
    id:'1',
    name:'2',
    code:'3',
    codeValue:''
};
let othObj = {
    id:'1',
    name:'2',
    code:'3',
    codeValue:'wct'
};
let needCheckAttrs = ['codeValue']
let had = isObjectValEqual(srcObj,othObj,needCheckAttrs);

/**
* 输出：
* had = true
*
*/

```
